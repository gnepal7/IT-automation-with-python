    ensure => directory,
  }

class timezone {
  file { '/etc/timezone':
    ensure  => file,
    content => "UTC\n",
    replace => true,
  }
}

# Puppet Classes
NTP :- Network Time Protocol
class ntp {
  package { 'ntp':
    ensure => latest,
  }
  file { '/etc/ntp.conf':
    source => 'puppet:///modules/ntp/ntp.conf'
    replace => true,
  }
  service { 'ntp':
    enable  => true,
    ensure  => running,
  }
}
